\newcommand{\importfiles}[1]{
  \getargs{#1}
  \setcounter{index}{0}
  \whiledo{\theindex < \narg}{
    \stepcounter{index}
    \importfile{\theindex}{\csname arg\romannumeral\theindex\endcsname}
  }
}
